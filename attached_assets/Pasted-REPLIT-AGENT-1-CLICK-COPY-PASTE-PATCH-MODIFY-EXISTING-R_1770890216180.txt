import React, { useEffect, useMemo, useRef, useState } from "react";

type Props = {
  disabled?: boolean;
  placeholder?: string;
  onSend: (text: string) => Promise<void> | void;
  autoSpeakText?: string;
  defaultTtsEnabled?: boolean;
  defaultMicMode?: "toggle" | "push";
};

declare global {
  interface Window {
    webkitSpeechRecognition?: any;
    SpeechRecognition?: any;
  }
}

export default function SupremeComposer({
  disabled = false,
  placeholder = "Message Supreme Assistant…",
  onSend,
  autoSpeakText,
  defaultTtsEnabled = false,
  defaultMicMode = "toggle",
}: Props) {
  const [value, setValue] = useState("");
  const [isListening, setIsListening] = useState(false);
  const [ttsEnabled, setTtsEnabled] = useState(defaultTtsEnabled);
  const [micMode, setMicMode] = useState<"toggle" | "push">(defaultMicMode);
  const [lastSpokenHash, setLastSpokenHash] = useState<string>("");

  const textareaRef = useRef<HTMLTextAreaElement | null>(null);
  const recognitionRef = useRef<any | null>(null);

  const sttSupported = useMemo(
    () => Boolean(window.SpeechRecognition || window.webkitSpeechRecognition),
    []
  );

  const ttsSupported = useMemo(
    () => typeof window !== "undefined" && "speechSynthesis" in window && "SpeechSynthesisUtterance" in window,
    []
  );

  // Auto-grow textarea
  useEffect(() => {
    const el = textareaRef.current;
    if (!el) return;
    el.style.height = "0px";
    const next = Math.min(el.scrollHeight, 180);
    el.style.height = `${next}px`;
  }, [value]);

  // Setup STT
  useEffect(() => {
    if (!sttSupported) return;

    const SR = window.SpeechRecognition || window.webkitSpeechRecognition;
    const rec = new SR();
    rec.continuous = false;
    rec.interimResults = true;
    rec.lang = "en-US";

    let finalTranscript = "";

    rec.onstart = () => {
      finalTranscript = "";
      setIsListening(true);
    };

    rec.onerror = (e: any) => {
      setIsListening(false);
      console.warn("[SupremeComposer] STT error:", e?.error || e);
    };

    rec.onend = () => {
      setIsListening(false);
      if (finalTranscript.trim()) {
        setValue((prev) => {
          const merged = (prev ? prev + " " : "") + finalTranscript.trim();
          return merged.replace(/\s+/g, " ").trim();
        });
      }
    };

    rec.onresult = (event: any) => {
      let interim = "";
      for (let i = event.resultIndex; i < event.results.length; i++) {
        const res = event.results[i];
        const txt = res[0]?.transcript ?? "";
        if (res.isFinal) finalTranscript += txt;
        else interim += txt;
      }
      const preview = (finalTranscript + interim).trim();
      if (preview) setValue(preview);
    };

    recognitionRef.current = rec;

    return () => {
      try { rec.stop(); } catch {}
      recognitionRef.current = null;
    };
  }, [sttSupported]);

  // TTS
  function stopSpeaking() {
    if (!ttsSupported) return;
    window.speechSynthesis.cancel();
  }

  function speak(text: string) {
    if (!ttsSupported || !text?.trim()) return;
    window.speechSynthesis.cancel();
    const u = new SpeechSynthesisUtterance(text);
    u.rate = 1;
    u.pitch = 1;
    u.volume = 1;
    window.speechSynthesis.speak(u);
  }

  // Auto-speak newest assistant message
  useEffect(() => {
    if (!ttsEnabled) return;
    if (!autoSpeakText?.trim()) return;

    const hash =
      String(autoSpeakText.length) +
      ":" +
      autoSpeakText.slice(0, 40) +
      ":" +
      autoSpeakText.slice(-40);

    if (hash === lastSpokenHash) return;

    speak(autoSpeakText);
    setLastSpokenHash(hash);
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [autoSpeakText, ttsEnabled]);

  // Mic controls
  function startListening() {
    if (!sttSupported || disabled) return;
    try { recognitionRef.current?.start(); } catch {}
  }
  function stopListening() {
    try { recognitionRef.current?.stop(); } catch {}
  }
  function toggleListening() {
    if (!sttSupported || disabled) return;
    if (isListening) stopListening();
    else startListening();
  }

  // Send
  async function doSend() {
    const text = value.trim();
    if (!text || disabled) return;
    setValue("");
    stopListening();
    await onSend(text);
    requestAnimationFrame(() => textareaRef.current?.focus());
  }

  function onKeyDown(e: React.KeyboardEvent<HTMLTextAreaElement>) {
    if (e.key === "Enter" && !e.shiftKey) {
      e.preventDefault();
      void doSend();
    }
  }

  function toggleTts() {
    const next = !ttsEnabled;
    setTtsEnabled(next);
    if (!next) stopSpeaking();
  }

  return (
    <div className="supreme-composer-wrap">
      <div className="supreme-composer">
        <textarea
          ref={textareaRef}
          className="supreme-input"
          value={value}
          onChange={(e) => setValue(e.target.value)}
          onKeyDown={onKeyDown}
          placeholder={placeholder}
          disabled={disabled}
          rows={1}
        />

        <div className="supreme-actions">
          <button
            type="button"
            className={[
              "supreme-iconbtn",
              "mic",
              isListening ? "active" : "",
              !sttSupported ? "disabled" : "",
            ].join(" ")}
            title={!sttSupported ? "Voice input not supported in this browser" : isListening ? "Stop listening" : "Voice input"}
            disabled={disabled || !sttSupported}
            onClick={micMode === "toggle" ? toggleListening : undefined}
            onMouseDown={micMode === "push" ? startListening : undefined}
            onMouseUp={micMode === "push" ? stopListening : undefined}
            onTouchStart={micMode === "push" ? startListening : undefined}
            onTouchEnd={micMode === "push" ? stopListening : undefined}
          >
            <span className="icon mic" aria-hidden="true" />
            {isListening && <span className="pulse" aria-hidden="true" />}
          </button>

          <button
            type="button"
            className={[
              "supreme-iconbtn",
              "speaker",
              ttsEnabled ? "active" : "",
              !ttsSupported ? "disabled" : "",
            ].join(" ")}
            title={!ttsSupported ? "Text-to-speech not supported" : ttsEnabled ? "Disable voice replies" : "Enable voice replies"}
            disabled={disabled || !ttsSupported}
            onClick={toggleTts}
          >
            <span className="icon speaker" aria-hidden="true" />
          </button>

          <button
            type="button"
            className={["supreme-send", value.trim() ? "ready" : ""].join(" ")}
            disabled={disabled || !value.trim()}
            onClick={() => void doSend()}
            title="Send"
          >
            <span className="icon send" aria-hidden="true" />
          </button>
        </div>
      </div>

      <div className="supreme-subrow">
        <button
          type="button"
          className="supreme-link"
          onClick={() => setMicMode((m) => (m === "toggle" ? "push" : "toggle"))}
          title="Switch mic mode"
        >
          Mic: {micMode === "toggle" ? "Tap to talk" : "Hold to talk"}
        </button>
      </div>
    </div>
  );
}
